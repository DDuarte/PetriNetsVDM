class DFSStack
  instance variables
    stack: seq of DFSElement := [];
  operations
(*@
\label{reset:5}
@*)
    public reset: () ==> ()
    reset() ==
      stack := [];
(*@
\label{pop:8}
@*)
    public pop: () ==> DFSElement
    pop() ==
      def res = hd stack in
      (stack := tl stack;
      return res)
    pre stack <> [];

(*@
\label{push:15}
@*)
    public push: DFSElement ==> ()
    push(element) == (
      stack := [element] ^ stack;
    );

(*@
\label{empty:20}
@*)
    public empty: () ==> bool
    empty() == (
      return len stack = 0
    );

end DFSStack
